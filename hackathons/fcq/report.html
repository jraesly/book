var grps = _.groupBy(data, 'CrsPBADept')
var ret1 = _.mapValues(grps, function(d){
  var enrollNumbers = _.pluck(d, 'N.ENROLL')
  return _.sum(enrollNumbers)
  })
  var min = _.min(ret1)
  var dept = _.pick(ret1, function(d){
    return d == min
    })
